package physicalform

import (
	"github.com/cowsed/ElfHouse/Helpers/Grammar"
	"fmt"
)

//All parameters must be copy by value basic types
type PhysicalForm struct {
	CommonName string //Human, dwarf, dog, cat etc

	HairColor   string //top of hair color for bipeds, body hair color for others
	AccentColor string //either like mustache color or mane color or spot color
	EyeColor    string

	Height float64 //in meters, assumes general dimensions of an animal
	Weight float64 //in kg

	WalkSpeed    uint8 //unit is decimeters per second (meters are too long) Extended jog would be 2x, sprint would be 6x
	JumpStrength uint8 //decimeters. lowest .1 meter highest 25 meters which is a lot

	Strength uint8 //General lifting strength

	//Flags
	Flammable bool
	Freezable bool

	//Resistance amounts, how much one can resist can be influenced by clothing or magic
	Lava uint8 //For a human 0 cuz humans cant go through lava
	Snow uint8 //For a human maybe 180 can get through some snow but not tons
	Heat uint8 //You get the point
	Cold uint8
	Acid uint8

	//Current Parameters
	MaxHealth int
	Health    int
}

func ( PhysicalForm) ( uint8) string {
	var  string
	if  == 0 {
		 = fmt.Sprintf("/Ps/ /tb/ %s strong. ", grammar.WordFromValue(.Strength))
	} else {
		//Base Physical aspects
		 = fmt.Sprintf("/Ps/ /tb/ %.1f meters tall and /tb/ %.2f kg. ", .Height, .Weight)
		//Strength
		 += fmt.Sprintf("/Ps/ /tb/ %s strong and can vertically jump %.2f meters. ", grammar.WordFromValue(.Strength), float64(.JumpStrength)/10.0)
		//Colors
		 += fmt.Sprintf("/Ps/ /tv/ %s hair and %s eyes. ", .HairColor, .EyeColor)
	}
	return 
}
